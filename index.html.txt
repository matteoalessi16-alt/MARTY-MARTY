  <script>
    const noBtn = document.getElementById("noBtn");
    const yesBtn = document.getElementById("yesBtn");
    const message = document.getElementById("message");

    const minDistance = 15;
    const triggerDistance = 140;

    function clamp(value, min, max) {
      return Math.max(min, Math.min(max, value));
    }

    function getButtonCenter(rect) {
      return {
        x: rect.left + rect.width / 2,
        y: rect.top + rect.height / 2
      };
    }

    document.addEventListener("mousemove", (event) => {
      const rect = noBtn.getBoundingClientRect();
      const center = getButtonCenter(rect);

      const dx = center.x - event.clientX;
      const dy = center.y - event.clientY;
      const distance = Math.hypot(dx, dy);

      if (distance < triggerDistance) {
        let moveX;
        let moveY;

        if (distance === 0) {
          moveX = minDistance;
          moveY = minDistance;
        } else {
          const escapeDistance = (triggerDistance - distance) + minDistance;
          moveX = (dx / distance) * escapeDistance;
          moveY = (dy / distance) * escapeDistance;
        }

        const newLeft = clamp(
          rect.left + moveX,
          0,
          window.innerWidth - rect.width
        );

        const newTop = clamp(
          rect.top + moveY,
          0,
          window.innerHeight - rect.height
        );

        noBtn.style.left = `${newLeft}px`;
        noBtn.style.top = `${newTop}px`;
      }
    });

    yesBtn.addEventListener("click", () => {
      message.textContent = "Lo sapevo. ❤️";
    });
  </script>
